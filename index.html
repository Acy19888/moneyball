    <div class="result" id="result"></div>


    <script>

        const playerData = {

            marktwert: 4.39, // Millionen

            durchschnittPunkte: 5.32,

            spieleGespielt: 18,

            gesamtSpiele: 19,

            sofascore: 6.87

        };


        document.querySelectorAll("input[type='range']").forEach(slider => {

            slider.addEventListener("input", (e) => {

                document.getElementById(`${e.target.id.replace('-weight', '')}-value`).textContent = e.target.value;

            });

        });


        function calculateScore() {

            const effizienzWeight = parseFloat(document.getElementById('effizienz-weight').value);

            const leistungWeight = parseFloat(document.getElementById('leistung-weight').value);

            const konsistenzWeight = parseFloat(document.getElementById('konsistenz-weight').value);

            const sofascoreWeight = parseFloat(document.getElementById('sofascore-weight').value);


            const totalWeight = effizienzWeight + leistungWeight + konsistenzWeight + sofascoreWeight;


            const effizienzScore = ((playerData.durchschnittPunkte * playerData.spieleGespielt) / playerData.marktwert) * (effizienzWeight / totalWeight);

            const leistungScore = (playerData.durchschnittPunkte / 8) * (leistungWeight / totalWeight);

            const konsistenzScore = (playerData.spieleGespielt / playerData.gesamtSpiele) * (konsistenzWeight / totalWeight);

            const sofascoreScore = (playerData.sofascore / 10) * (sofascoreWeight / totalWeight);


            const finalScore = (effizienzScore + leistungScore + konsistenzScore + sofascoreScore) * 100;


            const rating = finalScore > 90 ? "Exzellent" :

                           finalScore > 75 ? "Sehr gut" :

                           finalScore > 50 ? "Gut" : "Durchschnittlich";


            document.getElementById('result').innerHTML = `

                <strong>Finaler Score:</strong> ${finalScore.toFixed(2)}<br>

                <strong>Bewertung:</strong> ${rating}

            `;

        }
